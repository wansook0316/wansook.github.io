**_실버1_** : 분할 정복 문제이다.  
[백준(10830번) - 행렬 곱셈](https://www.acmicpc.net/problem/10830)

```c++
typedef vector<vector<int>> matrix;

matrix operator * (const matrix &a, const matrix &b) {
    int n = int(a.size());
    matrix ans(n, vector<int>(n));

    for (int i = 0; i < n; i++) {
        for (int j = 0; j < n; j++) {
            for (int k = 0; k < n; k++) {
                ans[i][j] += a[i][k] * b[k][j];
            }
            ans[i][j] %= 1000;
        }
    }
    return ans;
}

matrix go(matrix a, long long b){
    int n = int(a.size());
    matrix ans(n, vector<int>(n));

    if (b == 0) return ans;
    if (b % 2) ans = ans * a;
    return ans * go(a*a, b/2);
}



int main(){
    int n;
    long long b;
    cin >> n >> b;

    matrix a(n, vector<int>(n));

    for (int i = 0; i < n; i++)
        for (int j = 0; j < n; j++)
            cin >> a[i][j];

    matrix ans == calc(a, b);
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < n; j++) {
            cout << ans[i][j] << ' ';
        }cout << '\n';
    }


}
```
